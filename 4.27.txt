#include<stdio.h>
int Add(int x,int y)
{
	int z=0;
	z=x+y;
	return z;
}
int main()
{
	int a=10;
	int b=20;
	int (*p)(int,int)=Add;
    printf("%d\n",p(2,3));//函数解引用，这个*只是一个摆设。有没有，有几个都不影响引用函数。
    printf("%d\n",(*p)(2,3));//因为函数名也可以当作地址
	printf("%d\n",(**p)(2,3));//地址也可以当函数
	printf("%d\n",(***p)(2,3));

	printf("%d\n",Add(a,b));

	printf("%p\n",&Add);
	printf("%p\n",Add);//&Add与Add是一样的，不存在什么函数首元素地址
	printf("%p\n",p);
	return 0;
}





#include<stdio.h>
void Print(char* str)
{
	printf("%s\n",str);
}
int main()
{
	void(*p)(char*)=Print;
	(*p)("hello bit");//字符串导入指针，只把它当成常量字符的地址导入
	return 0;
}
